package dicision_control_statement_for_loop;

public class Forloop {

	public static void main(String[] args) {
		// sum of 10 numbers
	int sum=0;
	for(int i=1;i<=10;i++) {
	sum=sum+i;// sum+=i;
	System.out.println(sum);
	}

	}

}
